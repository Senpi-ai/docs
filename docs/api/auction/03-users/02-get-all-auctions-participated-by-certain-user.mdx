## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get all auctions participated by a certain user"
  subgraphLink={SUBGRAPH_URL.auction}
  query={`query MyQuery($userAddress: Bytes!) {
  users(where: { address: $userAddress }) {
    participatedAuctions {
      auctionId
      auctioningToken {
        id
        symbol
        decimals
      }
      txHash
    }
    address
  }
}`}
  variable={`{
  "userAddress": "0x1a45675bd474c8e3056a7dbd19e738d7849c0d26"
}`}
  response={`{
  "data": {
    "users": [
      {
        "participatedAuctions": [
          {
            "auctionId": "37",
            "auctioningToken": {
              "id": "0xf08c342445e8ca84476941078a109a496244187f",
              "symbol": "cid:goldy",
              "decimals": "18"
            },
            "txHash": "0x2a96c4e8032e8eb270f17f8e4487630b171be3c20a75e8547196378f54ffab5f"
          }
        ],
        "address": "0x1a45675bd474c8e3056a7dbd19e738d7849c0d26"
      }
    ]
  }
}`}
/>
