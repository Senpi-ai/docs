## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get all canceled bids from a certain auction"
  subgraphLink={SUBGRAPH_URL.auction}
  query={`query MyQuery($auctionId: BigInt!) {
  cancellationSellOrders(where: { auctionId: $auctionId }) {
    buyAmount
    sellAmount
    userId
    blockInfo {
      blockNumber
      hash
      timestamp
    }
  }
}`}
  variable={`{
  "auctionId": "37"
}`}
  response={`{
  "data": {
    "cancellationSellOrders": [
      {
        "buyAmount": "11000000000000000000",
        "sellAmount": "11000000000000000000",
        "userId": "9",
        "blockInfo": {
          "blockNumber": "12667487",
          "hash": "0x2b422395d0786933c689a17df179f2657a6844a53dc9e8b2e523993cff785921",
          "timestamp": "1721103262"
        }
      }
    ]
  }
}`}
/>
