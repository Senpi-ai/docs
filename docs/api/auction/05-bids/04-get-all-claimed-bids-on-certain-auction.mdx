## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get all claimed bids on a certain auction"
  subgraphLink={url.auction_stats_subgraphs}
  query={`query MyQuery($auctionId: BigInt!) {
  claimedFromOrders(where: { auctionId: $auctionId }) {
    auctionId
    buyAmount
    sellAmount
    userId
    blockInfo {
      blockNumber
      hash
      timestamp
    }
  }
}`}
  variable={`{
  "auctionId": "4"
}`}
  response={`{
  "data": {
    "claimedFromOrders": [
      {
        "auctionId": "4",
        "buyAmount": "1000000000000000000",
        "sellAmount": "4000000000000000000",
        "userId": "2",

          "timestamp": "1720266078"
        }
      }
    ]
  }
}`}
/>
