## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get all the fan tokens owned by a certain user"
  subgraphLink={SUBGRAPH_URL.protocol}
  query={`query MyQuery($userAddress: ID) {
  users(where: { id: $userAddress }) {
    portfolio {
      balance
      protocolTokenSpent
      subjectToken {
        name
        symbol
      }
    }
  }
}`}
  variable={`{
  "userAddress": "0x20e04003c2edfbca86300f6b2e6736142718dae3"
}`}
  response={`{
  "data": {
    "users": [
      {
        "portfolio": [
          {
            "balance": "772821289221015894850",
            "protocolTokenSpent": "500000000000000000000",
            "subject": {
              "name": "fid-3761",
              "symbol": "fid-3761"
            }
          }
        ]
      }
    ]
  }
}`}
/>
