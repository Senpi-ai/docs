## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get all the fan tokens owned by a certain user"
  subgraphLink={url.protocol_stats_subgraphs}
  query={
    `query MyQuery($userAddress: ID) {
  users(where: { id: $userAddress }) {
    portfolio {
      balance
      protocolTokenSpent
      subjectToken {
        name
        symbol
      }
    }
  }
}` ??
    `query MyQuery($userAddress: ID) {
  users(where: { id: $userAddress }) {
    portfolio {
      balance
      buyVolume
      sellVolume
      subjectToken {
        name
        symbol
      }
    }
  }
}`
  }
  variable={`{
  "userAddress": "0x00d620def6ccb76c92dbfc87bc2bebab7637ec53"
}`}
  response={
    `{
  "data": {
    "users": [
