## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get user's balance of a certain fan token"
  subgraphLink={SUBGRAPH_URL.protocol}
  query={`query MyQuery($userAddress: ID, $fanTokenAddress: String) {
  users(where: { id: $userAddress }) {
    portfolio(where: { subject: $fanTokenAddress }) {
      balance
      protocolTokenSpent
      subject {
        name
        symbol
      }
    }
  }
}`}
  variable={`{
  "userAddress": "0x20e04003c2edfbca86300f6b2e6736142718dae3",
  // provide the fan token address here to get user's balance
  "fanTokenAddress": "0xf07fbe409a57c4cf896ed24c5a438058a7abe39c"
}`}
  response={`{
  "data": {
    "users": [
      {
        "portfolio": [
          {
            "balance": "772821289221015894850",
            "protocolTokenSpent": "500000000000000000000",
            "subject": {
              "name": "fid-3761",
              "symbol": "fid-3761"
            }
          }
        ]
      }
    ]
  }
}`}
/>
