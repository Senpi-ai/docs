## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get user's balance of a certain fan token"
  subgraphLink={url.protocol_analytics_subgraphs}
  query={
    `query MyQuery($userAddress: ID, $fanTokenAddress: String) {
  users(where: { id: $userAddress }) {
    portfolio(where: { subjectToken: $fanTokenAddress }) {
      balance
      protocolTokenSpent
      subjectToken {
        name
        symbol
      }
    }
  }
}` ??
    `query MyQuery($userAddress: ID, $fanTokenAddress: String) {
  users(where: { id: $userAddress }) {
    portfolio(where: { subjectToken: $fanTokenAddress }) {
      balance
      buyVolume
      sellVolume
      subjectToken {
        name
        symbol
      }
    }
  }
}`
  }
  variable={`{
  "userAddress": "0x00d620def6ccb76c92dbfc87bc2bebab7637ec53",
  // provide the fan token address here to get user's balance
  "fanTokenAddress": "0x00e7d5fad22a375568d03b0688abbdfa4ec3b729"
}`}
