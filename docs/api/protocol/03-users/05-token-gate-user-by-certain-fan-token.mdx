## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="build token gate feature to limit access to a certain fan token"
  subgraphLink={SUBGRAPH_URL.protocol}
  query={`query MyQuery($userAddress: ID, $fanTokenName: String) {
  users(where: { id: $userAddress }) {
    portfolio(where: { subjectToken_: { name: $fanTokenName} }) {
      balance
    }
  }
}`}
  variable={`{
  "userAddress": "0x20e04003c2edfbca86300f6b2e6736142718dae3",
  /**
   * There are 3 types of fan tokens, here's their naming format:
   * - USER -> fid:<fid>
   * - CHANNEL -> cid:<channel-id>
   * - NETWORK (Only for Farcaster) -> id:farcaster
   */
  "fanTokenName": "fid:289309"
}`}
  response={`{
  "data": {
    "users": [
      {
        "portfolio": [
          {
            "balance": "501000000000000000000"
          }
        ]
      }
    ]
  }
}`}
  customLogic={`console.log(
    // Check if the user has a balance greater than 0
    Number.parseInt(data?.users?.[0]?.portfolio?.[0]?.balance) > 0
  );`}
/>
