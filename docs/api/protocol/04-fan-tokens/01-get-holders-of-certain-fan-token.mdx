## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="all the holders of a certain fan token"
  subgraphLink={url.protocol_stats_subgraphs}
  query={`query MyQuery($fanTokenAddress: ID) {
  subjectTokens(where: {id: $fanTokenAddress}) {
    portfolio {
      balance
      user {
        id
      }
    }
  }
}`}
  variable={`{
  "fanTokenAddress": "0x00e7d5fad22a375568d03b0688abbdfa4ec3b729"
}`}
  response={`{
  "data": {
    "subjectTokens": [
      {
        "portfolio": [
          {
            "balance": "10000000000000000000",
            "user": {
              "id": "0x00d620def6ccb76c92dbfc87bc2bebab7637ec53"
            }
          }
        ]
      }
    ]
  }
}`}
/>
