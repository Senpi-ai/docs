## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="all the holders of a certain fan token"
  subgraphLink={url.protocol_analytics_subgraphs}
  query={
    `query MyQuery($fanTokenAddress: ID) {
  subjectTokens(where: { id: $fanTokenAddress }) {
    holders {
      id
    }
  }
}` ??
    `query MyQuery($fanTokenAddress: ID) {
  subjectTokens(where: {id: $fanTokenAddress}) {
    portfolio {
      balance
      user {
        id
      }
    }
  }
}`
  }
  variable={`{
  "fanTokenAddress": "0x00e7d5fad22a375568d03b0688abbdfa4ec3b729"
}`}
