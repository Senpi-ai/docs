## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="all the holders of a certain fan token"
  subgraphLink={SUBGRAPH_URL.protocol}
  query={`query MyQuery($fanTokenAddress: ID) {
  subjects(where: { id: $fanTokenAddress }) {
    holders {
      id
    }
  }
}`}
  variable={`{
  "fanTokenAddress": "0x035eb3b4230b980723c71d3c09d6561824f1610b"
}`}
  response={`{
  "data": {
    "subjects": [
      {
        "holders": [
          {
            "id": "0x28e8a1817af21d39d6820fd930fde618e03d4d8d"
          },
          {
            "id": "0x3199219bda5dbc9ee45bb3dfad180be56a7ebfd0"
          },
          {
            "id": "0xe665cebf6111b6bcd8b84538d31f4048666b489a"
          }
        ]
      }
    ]
  }
}`}
/>
