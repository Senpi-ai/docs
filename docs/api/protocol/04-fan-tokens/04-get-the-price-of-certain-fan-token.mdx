## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get the price of a certain fan token"
  subgraphLink={SUBGRAPH_URL.protocol}
  query={`query MyQuery($fanTokenAddress: ID) {
  subjects(where: { id: $fanTokenAddress }) {
    currentPrice
  }
}`}
  variable={`{
  "fanTokenAddress": "0x035eb3b4230b980723c71d3c09d6561824f1610b"
}`}
  response={`{
  "data": {
    "subjects": [
      {
        "currentPrice": "0"
      }
    ]
  }
}`}
/>
