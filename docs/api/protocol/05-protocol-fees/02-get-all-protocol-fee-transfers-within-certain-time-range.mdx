## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get all the protocol fee transfers on the Moxie protocol within a certain time range"
  subgraphLink={url.protocol_analytics_subgraphs}
  query={`query MyQuery($startTime: BigInt, $endTime: BigInt) {
  protocolFeeTransfers(
    where: {
      blockInfo_: { timestamp_gte: $startTime, timestamp_lte: $endTime }
    }
  ) {
    amount
    beneficiary {
      beneficiary
      totalFees
    }
    txHash
    order {
      orderType
    }
    subjectToken {
      address: id
    }
  }
}`}
  variable={`{
  "startTime": "1720630000", // start timestamp
  "endTime": "1720640000" // end timestamp
}`}
  response={`{
