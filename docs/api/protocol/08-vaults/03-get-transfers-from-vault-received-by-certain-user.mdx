## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get all the Moxie transfers from the vault contract received by a certain user"
  subgraphLink={SUBGRAPH_URL.protocol}
  query={`query MyQuery($receiver: Bytes) {
  vaultTransferTxes(where: { to: $receiver }) {
    amount
    to
    subject
    token
    totalReserve
    txHash
    blockInfo {
      blockNumber
      hash
      timestamp
    }
  }
}`}
  variable={`{
  "receiver": "0x0f0ecb17575c232077075ef9d78fc9afad808b4d"
}`}
