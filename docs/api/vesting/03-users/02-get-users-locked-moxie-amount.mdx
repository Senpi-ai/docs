## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get the user's locked Moxie amount in the vesting contract"
  subgraphLink={url.vesting_subgraphs_legacy}
  query={`query MyQuery($userAddress: Bytes!) {
  tokenLockWallets(
    where: {beneficiary: $userAddress}
  ) {
    token
    tokensReleased
    tokensRevoked
    tokensWithdrawn
    tokenDestinationsApproved
  }
}`}
  variable={`{
  "userAddress": "0xdb48314a86839b04c3434575cee9346f20d9070e"
}`}
  response={`{
  "data": {
    "tokenLockWallets": [
      {
        "token": "0xf86136afb0fb72cdbd27d7aea9e283725d6815a1",
        "tokensReleased": "0",
        "tokensRevoked": "0",
        "tokensWithdrawn": "0",
        "tokenDestinationsApproved": false
      }
    ]
  }
}`}
/>
