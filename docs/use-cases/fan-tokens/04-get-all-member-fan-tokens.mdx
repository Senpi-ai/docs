## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get all the member-type fan tokens in the Moxie protocol"
  subgraphLink={url.protocol_stats_subgraphs}
  query={`query MyQuery($symbol_starts_with: String) {
  subjectTokens(where: {symbol_starts_with: $symbol_starts_with}) {
    address: id
    name
    symbol
    decimals
  }
}`}
  variable={`{
  // Check if the symbol starts with "fid:" to include only member-type fan tokens
  "symbol_starts_with": "fid:"
}`}
  response={`{
  "data": {
    "subjectTokens": [
      {
        "address": "0x014ec746d007d0683d284218f687c161b070dc05",
        "name": "ted",
        "symbol": "fid:239",
        "decimals": "18"
      }
    ]
  }
}`}
/>
