## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get all the holders of a certain Creator Coin"
  subgraphLink={url.protocol_stats_subgraphs}
  query={`query MyQuery($fanTokenSymbol: String) {
  subjectTokens(where: {symbol: $fanTokenSymbol}) {
    portfolio(
      where: {
        balance_gt: 0,
        # Exclude the staking contract
        user_not_in: ["0xcb2513d389354f7ac80f5042bb8948a234a439b2"]
      }
    ) {
      stakedBalance
      unstakedBalance
      user {
        id
      }
    }
  }
}`}
  variable={`{
  "fanTokenSymbol": "fid:3"
}`}
  response={`{
  "data": {
    "subjectTokens": [
      {
        "portfolio": [
          {
            "stakedBalance": "0",
            "unstakedBalance": "1214949183764430400",
            "user": {
              "id": "0x00000000009726632680fb29d3f7a9734e3010e2"
            }
          },
          // Other holders of FID 3
        ]
      }
    ]
  }
}`}
/>
