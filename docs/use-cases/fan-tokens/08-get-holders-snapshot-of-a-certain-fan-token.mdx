## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get all the holders snapshot of a certain fan token"
  subgraphLink={url.protocol_stats_subgraphs}
  query={`query MyQuery($symbol: String, $blockNumber: Int) {
  subjectTokens(where: {symbol: $symbol}, block: {number: $blockNumber}) {
    portfolio(where: {balance_gt: 0}) {
      balance
      user {
        id
      }
    }
  }
}`}
  variable={`{
  /**
   * There are 3 types of fan tokens, here's how the symbol is formatted:
   * - USER -> fid:<fid>
   * - CHANNEL -> cid:<channel-id>
   * - NETWORK (Only for Farcaster) -> id:farcaster
   */
  "symbol": "id:farcaster",
  "blockNumber": 17895057
}`}
  response={`{
  "data": {
    "subjectTokens": [
      {
        "portfolio": [
          {
            "balance": "11616135683624072140",
            "user": {
              "id": "0x011c08baf3908843c30391767890d7e9a94116fe"
            }
          }
        ]
      }
    ]
  }
}`}
/>
