## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get the member fan token address by a certain FID"
  subgraphLink={url.protocol_stats_subgraphs}
  query={`query MyQuery($fanTokenSymbol: String) {
  subjectTokens(where: {symbol: $fanTokenSymbol}) {
    tokenAddress: id
  }
}`}
  variable={`{
  // Format the fan token symbol as "fid:<FID>"
  "fanTokenSymbol": "fid:12"
}`}
  response={`{
  "data": {
    "subjectTokens": [
      {
        "tokenAddress": "0x3b433d547005ab9fc97589e921ae0491198e6883"
      }
    ]
  }
}`}
/>
