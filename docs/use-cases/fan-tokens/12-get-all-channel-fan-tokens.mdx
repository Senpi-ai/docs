## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get all the channel-type Creator Coins in the Moxie protocol"
  subgraphLink={url.protocol_stats_subgraphs}
  query={`query MyQuery($symbol_starts_with: String) {
  subjectTokens(where: {symbol_starts_with: $symbol_starts_with}) {
    address: id
    name
    symbol
    decimals
  }
}`}
  variable={`{
  // Check if the symbol starts with "cid:" to include only channel-type Creator Coins
  "symbol_starts_with": "cid:"
}`}
  response={`{
  "data": {
    "subjectTokens": [
      {
        "address": "0x00e7d5fad22a375568d03b0688abbdfa4ec3b729",
        "name": "/fitness",
        "symbol": "cid:fitness",
        "decimals": "18"
      },
    ]
  }
}`}
/>
