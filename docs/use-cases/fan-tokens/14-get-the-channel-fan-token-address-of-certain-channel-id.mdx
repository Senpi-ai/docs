## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get the channel fan token address by a certain channel ID"
  subgraphLink={url.protocol_stats_subgraphs}
  query={`query MyQuery($fanTokenSymbol: String) {
  subjectTokens(where: {symbol: $fanTokenSymbol}) {
    tokenAddress: id
  }
}`}
  variable={`{
  // Format the fan token symbol as "cid:<channel-ID>"
  "fanTokenSymbol": "cid:airstack"
}`}
  response={`{
  "data": {
    "subjectTokens": [
      {
        "tokenAddress": "0x3a2281e71dc0aabecfa8045959cf3020f2a562e6"
      }
    ]
  }
}`}
/>
