## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get the hourly snapshots of a certain fan token"
  subgraphLink={url.protocol_stats_subgraphs}
  query={`query MyQuery($fanTokenAddress: ID) {
  subjectTokens(where: { id: $fanTokenAddress }) {
    hourlySnapshots {
      hourlyPriceChange
      hourlyProtocolFeeChange
      hourlyUniqueHoldersChange
      hourlyVolumeChange
      endPrice
      endProtocolFee
      endTimestamp
      endUniqueHolders
      endVolume
      reserve
      startPrice
      startProtocolFee
      startUniqueHolders
      startVolume
      totalSupply
    }
  }
}`}
  variable={`{
  "fanTokenAddress": "0x0045214aa342f75ff65c0208002302b57ddb9c22"
}`}
