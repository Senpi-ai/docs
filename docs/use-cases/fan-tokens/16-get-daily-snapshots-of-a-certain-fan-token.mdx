## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get all the daily snapshots of certain Creator Coin"
  subgraphLink={url.protocol_stats_subgraphs}

  subjectTokens(where: { id: $fanTokenAddress }) {
    dailySnapshots {
      dailyPriceChange
      dailyProtocolFeeChange
      dailyUniqueHoldersChange
      dailyVolumeChange
      endPrice
      endProtocolFee
      endTimestamp
      endUniqueHolders
      endVolume
      reserve
      startPrice
      startProtocolFee
      startUniqueHolders
      startVolume
      totalSupply
    }
  }
}`}
