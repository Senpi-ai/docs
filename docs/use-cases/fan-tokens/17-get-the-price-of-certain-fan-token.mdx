## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get the price of a certain Creator Coin"
  subgraphLink={url.protocol_stats_subgraphs}
  query={`query MyQuery($fanTokenAddress: ID) {
  subjectTokens(where: { id: $fanTokenAddress }) {
    currentPriceInMoxie
  }
}`}
  variable={`{
  "fanTokenAddress": "0x0045214aa342f75ff65c0208002302b57ddb9c22"
}`}
  response={`{
  "data": {
    "subjectTokens": [
      {
        "currentPriceInMoxie": "0.01536057923231617336512979146637123"
      }
    ]
  }
}`}
/>
