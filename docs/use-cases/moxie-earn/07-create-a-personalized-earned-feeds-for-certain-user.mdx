# Pre-requisites

1. Install dependencies `graphql`and`graphql-request` to your project :

<Install dependencies="graphql graphql-request" />

2. Your will need a special API key to access to the Moxie Partner API. To request access, please fill in this form [here](https://forms.gle/eRQ6uiCzica2J6qo8).

## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="create a personalized earned feeds for a certain Farcaster user"
  subgraphLink="https://api.moxie.xyz/promotion/graphql"
  query={`query ($earnerFid: String!, $status: [String!]!) {
  FetchEligiblePromotions(input: {
    earnerFid: $earnerFid,
    status: $status,
  }) {
    earnerFid
    promotions {
      castHash
      castURL
      castedByFid
      createdAt
      promoterFid
      rewardBoost
      status
      updatedAt
      promotionId
      targetAudienceStatus
    }
  }
}`}
  headers={`{
  // Add Airstack API key here
  "x-airstack-promotion": process.env.AIRSTACK_API_KEY as string,
}`}
  variable={`{
  // specify the FID of the Farcaster user
  "earnerFid": "602",
  "status": "ACTIVE"],
}`}
  response={`{
  "data": {
    "FetchEligiblePromotions": {
      "earnerFid": "602",
      "promotions": [
        {

          "castedByFid": "290249",
          "createdAt": "2024-11-05 03:27:55.000",
          "promoterFid": "8408",

          "updatedAt": "2024-11-05 03:30:28.000",
          "promotionId": "a1bb3d5a-3757-4be3-baf5-c3755ef99542",

}`}
/>
