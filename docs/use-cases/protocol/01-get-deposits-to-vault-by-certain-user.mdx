## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get all the Moxie deposits to the vault contract done by a certain user"
  subgraphLink={url.protocol_stats_subgraphs}
  query={`query MyQuery($sender: Bytes) {
  vaultDepositTxes(where: { sender: $sender }) {
    amount
    sender
    subject
    token
    totalReserve
    txHash
    blockInfo {
      blockNumber
      hash
      timestamp
    }
  }
}`}
  variable={`{
  "sender": "0x0f0ecb17575c232077075ef9d78fc9afad808b4d"
}`}
  response={`{
  "data": {
    "vaultDepositTxes": [
