## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get all the Moxie deposits to the vault contract by a certain fan token"
  subgraphLink={url.protocol_stats_subgraphs}
  query={`query MyQuery($fanTokenAddress: Bytes) {
  vaultDepositTxes(where: { subject: $fanTokenAddress }) {
    amount
    sender
    subject
    token
    totalReserve
    txHash
    blockInfo {
      blockNumber
      hash
      timestamp
    }
  }
}`}
  variable={`{
  "fanTokenAddress": "0xf07fbe409a57c4cf896ed24c5a438058a7abe39c"
}`}
  response={`{
  "data": {
    "vaultDepositTxes": [
      {
        "amount": "1003000000000000000000",
        "sender": "0x0f0ecb17575c232077075ef9d78fc9afad808b4d",
