## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get all the Moxie transfers from the vault contract of a certain Creator Coin on the Moxie protocol"
  subgraphLink={url.protocol_stats_subgraphs}
  query={`query MyQuery($fanTokenAddress: Bytes) {
  vaultTransferTxes(where: { subject: $fanTokenAddress }) {
    amount
    to
    subject
    token
    totalReserve
    txHash
    blockInfo {
      blockNumber
      hash
      timestamp
    }
  }
}`}
  variable={`{
  "fanTokenAddress": "0xf07fbe409a57c4cf896ed24c5a438058a7abe39c"
}`}
  response={`{
