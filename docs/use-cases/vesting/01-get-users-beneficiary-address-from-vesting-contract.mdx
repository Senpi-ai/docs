## Step 1: Add The API Query To Your Code

<AddQuery
  useCase="get user's beneficiary address from vesting contract"
  subgraphLink={url.vesting_subgraphs}
  query={`query MyQuery($address: ID) {
  tokenLockWallets(where: {id: $address}) {
    beneficiary
    address: id
  }
}`}
  variable={`{
  "address": "0x0000f530fc6397d28deb9eb28890cac2afa5c893"
}`}
  response={`{
  "data": {
    "tokenLockWallets": [
      {
        "beneficiary": "0x9a15cdf8cfcc7b5c4bba01f1585be80bb3c3fe70",
        "address": "0x0000f530fc6397d28deb9eb28890cac2afa5c893"
      }
    ]
  }
}`}
/>
